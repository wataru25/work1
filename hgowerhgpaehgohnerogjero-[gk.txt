Option Explicit

Private Const RECORD_SHEET_NAME As String = "記録シート"
Private Const MAX_CHANGE_DETAIL_COUNT As Long = 10

'==================================================
' 入口処理
'==================================================
Public Sub CreateOrUpdateValidationFile()

    Dim wbCalc As Workbook
    Dim wbValidation As Workbook
    
    Dim outputFolderPath As String
    Dim validationFilePath As String
    Dim validationFileExists As Boolean
    
    Dim prevCalc As XlCalculation
    Dim prevScreenUpdating As Boolean
    Dim prevDisplayAlerts As Boolean
    Dim prevEnableEvents As Boolean
    
    On Error GoTo ErrHandler
    
    Set wbCalc = ThisWorkbook
    
    If Len(wbCalc.Path) = 0 Then
        Err.Raise vbObjectError + 1000, , "計測ファイルが未保存です。先に保存してください。"
    End If
    
    outputFolderPath = GetOutputFolderPath()
    ValidateFolderPath outputFolderPath
    
    validationFilePath = BuildValidationFilePath(wbCalc, outputFolderPath)
    validationFileExists = (Dir(validationFilePath) <> "")
    
    SaveApplicationState prevCalc, prevScreenUpdating, prevDisplayAlerts, prevEnableEvents
    SetApplicationForProcessing
    
    If validationFileExists Then
        UpdateExistingValidationFile wbCalc, validationFilePath
    Else
        CreateNewValidationFile wbCalc, validationFilePath
    End If
    
    MsgBox "処理が完了しました。" & vbCrLf & validationFilePath, vbInformation
    
ExitPoint:
    RestoreApplicationState prevCalc, prevScreenUpdating, prevDisplayAlerts, prevEnableEvents
    Exit Sub

ErrHandler:
    MsgBox "エラーが発生しました。" & vbCrLf & _
           "番号: " & Err.Number & vbCrLf & _
           "内容: " & Err.Description, vbCritical
    Resume ExitPoint

End Sub

'==================================================
' 新規作成 / 更新
'==================================================
Private Sub CreateNewValidationFile(ByVal wbCalc As Workbook, ByVal validationFilePath As String)

    Dim wbValidation As Workbook
    
    wbCalc.SaveCopyAs validationFilePath
    
    Set wbValidation = OpenWorkbookWithoutRecalculation(validationFilePath)
    
    ConvertAllSheetsToValues wbValidation
    WriteRecordSheetForNewFile wbValidation, wbCalc.fullName
    
    wbValidation.Save
    wbValidation.Close SaveChanges:=False

End Sub

Private Sub UpdateExistingValidationFile(ByVal wbCalc As Workbook, ByVal validationFilePath As String)

    Dim wbValidation As Workbook
    
    If IsWorkbookOpenByFullName(validationFilePath) Then
        Err.Raise vbObjectError + 1003, , "検証ファイルが既に開かれています。" & vbCrLf & validationFilePath
    End If
    
    Set wbValidation = OpenWorkbookWithoutRecalculation(validationFilePath)
    
    UpdateValidationSheets wbCalc, wbValidation
    
    wbValidation.Save
    wbValidation.Close SaveChanges:=False

End Sub

Private Sub UpdateValidationSheets(ByVal wbCalc As Workbook, ByVal wbValidation As Workbook)

    Dim wsCalc As Worksheet
    Dim wsValidation As Worksheet
    Dim wsRecord As Worksheet
    
    Dim updatedCells As Collection
    Dim nextRow As Long
    
    Set wsRecord = GetOrCreateRecordSheet(wbValidation)
    InitializeRecordSheetForUpdate wsRecord, wbCalc.fullName
    
    nextRow = 8
    
    For Each wsCalc In wbCalc.Worksheets
        
        If wsCalc.Name <> wsRecord.Name Then
            
            Set wsValidation = FindSheetByName(wbValidation, wsCalc.Name)
            
            If wsValidation Is Nothing Then
                CopyNewSheetToValidationFile wsCalc, wbValidation
                
                wsRecord.Cells(nextRow, 1).Value = wsCalc.Name
                wsRecord.Cells(nextRow, 2).Value = "新規追加"
                wsRecord.Cells(nextRow, 3).Value = "計測ファイルにある同名シートが検証ファイルになかったため、シートを追加しました。"
            Else
                Set updatedCells = GetChangedNumericCells(wsCalc, wsValidation)
                CopySheetAsValuesAndFormats wsCalc, wsValidation
                
                wsRecord.Cells(nextRow, 1).Value = wsCalc.Name
                
                If updatedCells.Count = 0 Then
                    wsRecord.Cells(nextRow, 2).Value = "更新なし"
                    wsRecord.Cells(nextRow, 3).Value = "-"
                Else
                    wsRecord.Cells(nextRow, 2).Value = "数値更新あり"
                    wsRecord.Cells(nextRow, 3).Value = BuildChangeSummary(updatedCells)
                End If
            End If
            
            nextRow = nextRow + 1
            
        End If
        
    Next wsCalc
    
    FormatRecordSheetUpdateTable wsRecord, 7, nextRow - 1
    FinalizeRecordSheet wsRecord

End Sub

Private Sub CopyNewSheetToValidationFile(ByVal wsCalc As Worksheet, ByVal wbValidation As Workbook)

    Dim wsNew As Worksheet
    
    wsCalc.Copy After:=wbValidation.Worksheets(wbValidation.Worksheets.Count)
    Set wsNew = wbValidation.Worksheets(wbValidation.Worksheets.Count)
    ConvertWorksheetToValues wsNew

End Sub

'==================================================
' パス・ブック操作
'==================================================
Private Function GetOutputFolderPath() As String

    Dim folderPath As String
    
    folderPath = Trim$(CStr(check_output.Range("output_path").Value))
    
    If Len(folderPath) = 0 Then
        Err.Raise vbObjectError + 1001, , "check_outputシートのoutput_pathが空です。"
    End If
    
    GetOutputFolderPath = folderPath

End Function

Private Sub ValidateFolderPath(ByVal folderPath As String)

    If Dir(folderPath, vbDirectory) = "" Then
        Err.Raise vbObjectError + 1002, , "出力先フォルダが存在しません。" & vbCrLf & folderPath
    End If

End Sub

Private Function BuildValidationFilePath(ByVal wbCalc As Workbook, ByVal outputFolderPath As String) As String

    Dim validationFileName As String
    
    validationFileName = "(検証用)" & wbCalc.Name
    
    If Right$(outputFolderPath, 1) = "\" Then
        BuildValidationFilePath = outputFolderPath & validationFileName
    Else
        BuildValidationFilePath = outputFolderPath & "\" & validationFileName
    End If

End Function

Private Function OpenWorkbookWithoutRecalculation(ByVal filePath As String) As Workbook

    Set OpenWorkbookWithoutRecalculation = Workbooks.Open( _
        fileName:=filePath, _
        UpdateLinks:=0, _
        ReadOnly:=False, _
        AddToMru:=False, _
        IgnoreReadOnlyRecommended:=True)

End Function

Private Function IsWorkbookOpenByFullName(ByVal fullName As String) As Boolean

    Dim wb As Workbook
    
    For Each wb In Application.Workbooks
        If StrComp(wb.fullName, fullName, vbTextCompare) = 0 Then
            IsWorkbookOpenByFullName = True
            Exit Function
        End If
    Next wb

End Function

'==================================================
' Excelの状態制御
'==================================================
Private Sub SaveApplicationState(ByRef prevCalc As XlCalculation, _
                                 ByRef prevScreenUpdating As Boolean, _
                                 ByRef prevDisplayAlerts As Boolean, _
                                 ByRef prevEnableEvents As Boolean)

    prevCalc = Application.Calculation
    prevScreenUpdating = Application.ScreenUpdating
    prevDisplayAlerts = Application.DisplayAlerts
    prevEnableEvents = Application.EnableEvents

End Sub

Private Sub SetApplicationForProcessing()

    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    Application.EnableEvents = False
    Application.Calculation = xlCalculationManual

End Sub

Private Sub RestoreApplicationState(ByVal prevCalc As XlCalculation, _
                                    ByVal prevScreenUpdating As Boolean, _
                                    ByVal prevDisplayAlerts As Boolean, _
                                    ByVal prevEnableEvents As Boolean)

    On Error Resume Next
    Application.Calculation = prevCalc
    Application.ScreenUpdating = prevScreenUpdating
    Application.DisplayAlerts = prevDisplayAlerts
    Application.EnableEvents = prevEnableEvents
    On Error GoTo 0

End Sub

'==================================================
' シート検索
'==================================================
Private Function GetOrCreateRecordSheet(ByVal wb As Workbook) As Worksheet

    Dim ws As Worksheet
    
    Set ws = FindSheetByName(wb, RECORD_SHEET_NAME)
    
    If ws Is Nothing Then
        Set ws = wb.Worksheets.Add(After:=wb.Worksheets(wb.Worksheets.Count))
        ws.Name = RECORD_SHEET_NAME
    End If
    
    Set GetOrCreateRecordSheet = ws

End Function

Private Function FindSheetByName(ByVal wb As Workbook, ByVal sheetName As String) As Worksheet

    Dim ws As Worksheet
    
    For Each ws In wb.Worksheets
        If StrComp(Trim$(ws.Name), Trim$(sheetName), vbTextCompare) = 0 Then
            Set FindSheetByName = ws
            Exit Function
        End If
    Next ws
    
    Set FindSheetByName = Nothing

End Function

'==================================================
' 値化・貼り付け
'==================================================
Private Sub ConvertAllSheetsToValues(ByVal wb As Workbook)

    Dim ws As Worksheet
    
    For Each ws In wb.Worksheets
        ConvertWorksheetToValues ws
    Next ws

End Sub

Private Sub ConvertWorksheetToValues(ByVal ws As Worksheet)

    Dim targetRange As Range
    
    Set targetRange = GetUsedDataRange(ws)
    If targetRange Is Nothing Then Exit Sub
    
    targetRange.Value = targetRange.Value

End Sub

Private Sub CopySheetAsValuesAndFormats(ByVal wsCalc As Worksheet, ByVal wsValidation As Worksheet)

    Dim srcRange As Range
    Dim dstRange As Range
    
    Set srcRange = GetUsedDataRange(wsCalc)
    
    wsValidation.Cells.Clear
    
    If srcRange Is Nothing Then Exit Sub
    
    CopyColumnWidths wsCalc, wsValidation
    CopyRowHeights wsCalc, wsValidation
    
    Set dstRange = wsValidation.Range(wsValidation.Cells(1, 1), _
                                      wsValidation.Cells(srcRange.Rows.Count, srcRange.Columns.Count))
    
    dstRange.Value = srcRange.Value
    
    srcRange.Copy
    dstRange.PasteSpecial Paste:=xlPasteFormats
    dstRange.PasteSpecial Paste:=xlPasteValidation
    dstRange.PasteSpecial Paste:=xlPasteComments
    Application.CutCopyMode = False

End Sub

Private Function GetUsedDataRange(ByVal ws As Worksheet) As Range

    Dim lastRow As Long
    Dim lastCol As Long
    
    lastRow = GetLastUsedRow(ws)
    lastCol = GetLastUsedCol(ws)
    
    If lastRow = 0 Or lastCol = 0 Then
        Set GetUsedDataRange = Nothing
    Else
        Set GetUsedDataRange = ws.Range(ws.Cells(1, 1), ws.Cells(lastRow, lastCol))
    End If

End Function

'==================================================
' 差分取得
'==================================================
Private Function GetChangedNumericCells(ByVal wsCalc As Worksheet, ByVal wsValidation As Worksheet) As Collection

    Dim result As New Collection
    
    Dim maxRow As Long
    Dim maxCol As Long
    Dim arrCalc As Variant
    Dim arrValidation As Variant
    
    Dim r As Long
    Dim c As Long
    
    maxRow = Application.Max(GetLastUsedRow(wsCalc), GetLastUsedRow(wsValidation))
    maxCol = Application.Max(GetLastUsedCol(wsCalc), GetLastUsedCol(wsValidation))
    
    If maxRow = 0 Or maxCol = 0 Then
        Set GetChangedNumericCells = result
        Exit Function
    End If
    
    arrCalc = wsCalc.Range(wsCalc.Cells(1, 1), wsCalc.Cells(maxRow, maxCol)).Value2
    arrValidation = wsValidation.Range(wsValidation.Cells(1, 1), wsValidation.Cells(maxRow, maxCol)).Value2
    
    For r = 1 To maxRow
        For c = 1 To maxCol
            
            If IsNumericCellValue(arrCalc(r, c)) And IsNumericCellValue(arrValidation(r, c)) Then
                If CDbl(arrCalc(r, c)) <> CDbl(arrValidation(r, c)) Then
                    result.Add wsCalc.Cells(r, c).Address(False, False) & _
                               " : " & arrValidation(r, c) & " → " & arrCalc(r, c)
                End If
                
            ElseIf IsNumericCellValue(arrCalc(r, c)) Xor IsNumericCellValue(arrValidation(r, c)) Then
                result.Add wsCalc.Cells(r, c).Address(False, False) & _
                           " : " & ValueToText(arrValidation(r, c)) & " → " & ValueToText(arrCalc(r, c))
            End If
            
        Next c
    Next r
    
    Set GetChangedNumericCells = result

End Function

Private Function BuildChangeSummary(ByVal updatedCells As Collection) As String

    Dim i As Long
    Dim textBody As String
    
    If updatedCells Is Nothing Or updatedCells.Count = 0 Then
        BuildChangeSummary = "-"
        Exit Function
    End If
    
    If updatedCells.Count <= MAX_CHANGE_DETAIL_COUNT Then
        For i = 1 To updatedCells.Count
            If i > 1 Then textBody = textBody & vbLf
            textBody = textBody & "・" & updatedCells(i)
        Next i
        
        BuildChangeSummary = CStr(updatedCells.Count) & "セル更新" & vbLf & textBody
    Else
        For i = 1 To MAX_CHANGE_DETAIL_COUNT
            If i > 1 Then textBody = textBody & vbLf
            textBody = textBody & "・" & updatedCells(i)
        Next i
        
        BuildChangeSummary = CStr(updatedCells.Count) & "セル更新" & vbLf & _
                             "先頭" & MAX_CHANGE_DETAIL_COUNT & "件を表示" & vbLf & _
                             textBody & vbLf & _
                             "・ほか " & CStr(updatedCells.Count - MAX_CHANGE_DETAIL_COUNT) & "件"
    End If

End Function

Private Function IsNumericCellValue(ByVal v As Variant) As Boolean

    If IsError(v) Then Exit Function
    If IsEmpty(v) Then Exit Function
    
    If VarType(v) = vbString Then
        If Trim$(CStr(v)) = "" Then Exit Function
    End If
    
    IsNumericCellValue = IsNumeric(v)

End Function

Private Function ValueToText(ByVal v As Variant) As String

    If IsError(v) Then
        ValueToText = "#ERROR"
    ElseIf IsEmpty(v) Then
        ValueToText = "(空白)"
    ElseIf VarType(v) = vbString And Trim$(CStr(v)) = "" Then
        ValueToText = "(空白)"
    Else
        ValueToText = CStr(v)
    End If

End Function

'==================================================
' 記録シート出力
'==================================================
Private Sub WriteRecordSheetForNewFile(ByVal wbValidation As Workbook, ByVal calcFileFullName As String)

    Dim wsRecord As Worksheet
    
    Set wsRecord = GetOrCreateRecordSheet(wbValidation)
    
    wsRecord.Cells.Clear
    
    WriteRecordSheetHeader wsRecord, "検証ファイル 作成記録"
    WriteRecordSheetMeta wsRecord, calcFileFullName, "新規作成"
    
    wsRecord.Range("A7").Value = "備考"
    wsRecord.Range("B7").Value = "新規作成時はシート一覧を作成しません。"
    
    FormatRecordSheetCommon wsRecord
    FormatRecordSheetMetaArea wsRecord
    FinalizeRecordSheet wsRecord

End Sub

Private Sub InitializeRecordSheetForUpdate(ByVal wsRecord As Worksheet, ByVal calcFileFullName As String)

    wsRecord.Cells.Clear
    
    WriteRecordSheetHeader wsRecord, "検証ファイル 更新記録"
    WriteRecordSheetMeta wsRecord, calcFileFullName, "更新"
    
    wsRecord.Range("A7").Value = "シート名"
    wsRecord.Range("B7").Value = "状態"
    wsRecord.Range("C7").Value = "変更内容"
    
    FormatRecordSheetCommon wsRecord
    FormatRecordSheetMetaArea wsRecord

End Sub

Private Sub WriteRecordSheetHeader(ByVal wsRecord As Worksheet, ByVal titleText As String)

    With wsRecord.Range("A1:E1")
        .Merge
        .Value = titleText
        .Font.Bold = True
        .Font.Size = 16
        .Interior.Color = RGB(31, 78, 121)
        .Font.Color = RGB(255, 255, 255)
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
        .RowHeight = 24
    End With

End Sub

Private Sub WriteRecordSheetMeta(ByVal wsRecord As Worksheet, ByVal calcFileFullName As String, ByVal processType As String)

    wsRecord.Range("A3").Value = "計測ファイルのパス"
    wsRecord.Range("A4").Value = "処理日時"
    wsRecord.Range("A5").Value = "処理区分"
    
    wsRecord.Range("B3").Value = calcFileFullName
    wsRecord.Range("B4").Value = Now
    wsRecord.Range("B5").Value = processType

End Sub

Private Sub FormatRecordSheetCommon(ByVal wsRecord As Worksheet)

    With wsRecord.Cells
        .Font.Name = "Meiryo UI"
        .Font.Size = 10
        .VerticalAlignment = xlCenter
    End With
    
    wsRecord.Columns("A").ColumnWidth = 22
    wsRecord.Columns("B").ColumnWidth = 16
    wsRecord.Columns("C").ColumnWidth = 100
    
    wsRecord.Columns("C").WrapText = True

End Sub

Private Sub FormatRecordSheetMetaArea(ByVal wsRecord As Worksheet)

    With wsRecord.Range("A3:A5")
        .Font.Bold = True
        .Interior.Color = RGB(217, 225, 242)
        .Borders.LineStyle = xlContinuous
    End With
    
    With wsRecord.Range("B3:B5")
        .Borders.LineStyle = xlContinuous
    End With
    
    wsRecord.Range("B4").NumberFormatLocal = "yyyy/mm/dd hh:mm:ss"

End Sub

Private Sub FormatRecordSheetUpdateTable(ByVal wsRecord As Worksheet, ByVal headerRow As Long, ByVal lastRow As Long)

    If lastRow < headerRow Then Exit Sub
    
    With wsRecord.Range(wsRecord.Cells(headerRow, 1), wsRecord.Cells(headerRow, 3))
        .Font.Bold = True
        .Interior.Color = RGB(68, 114, 196)
        .Font.Color = RGB(255, 255, 255)
        .HorizontalAlignment = xlCenter
        .Borders.LineStyle = xlContinuous
    End With
    
    With wsRecord.Range(wsRecord.Cells(headerRow + 1, 1), wsRecord.Cells(lastRow, 3))
        .Borders.LineStyle = xlContinuous
        .VerticalAlignment = xlTop
    End With
    
    ApplyStatusColors wsRecord, headerRow + 1, lastRow
    wsRecord.Range(wsRecord.Cells(headerRow, 1), wsRecord.Cells(lastRow, 3)).AutoFilter

End Sub

Private Sub ApplyStatusColors(ByVal wsRecord As Worksheet, ByVal startRow As Long, ByVal lastRow As Long)

    Dim r As Long
    Dim statusText As String
    
    For r = startRow To lastRow
        statusText = CStr(wsRecord.Cells(r, 2).Value)
        
        Select Case statusText
            Case "更新なし"
                wsRecord.Cells(r, 2).Interior.Color = RGB(226, 239, 218)
            Case "数値更新あり"
                wsRecord.Cells(r, 2).Interior.Color = RGB(255, 242, 204)
            Case "新規追加"
                wsRecord.Cells(r, 2).Interior.Color = RGB(221, 235, 247)
        End Select
    Next r

End Sub

Private Sub FinalizeRecordSheet(ByVal wsRecord As Worksheet)

    wsRecord.Columns("A:C").AutoFit
    wsRecord.Columns("C").ColumnWidth = 100
    wsRecord.Rows.AutoFit

End Sub

'==================================================
' 最終使用行列
'==================================================
Private Function GetLastUsedRow(ByVal ws As Worksheet) As Long

    Dim lastCell As Range
    
    On Error Resume Next
    Set lastCell = ws.Cells.Find(What:="*", _
                                 After:=ws.Cells(1, 1), _
                                 LookIn:=xlFormulas, _
                                 LookAt:=xlPart, _
                                 SearchOrder:=xlByRows, _
                                 SearchDirection:=xlPrevious)
    On Error GoTo 0
    
    If lastCell Is Nothing Then
        GetLastUsedRow = 0
    Else
        GetLastUsedRow = lastCell.Row
    End If

End Function

Private Function GetLastUsedCol(ByVal ws As Worksheet) As Long

    Dim lastCell As Range
    
    On Error Resume Next
    Set lastCell = ws.Cells.Find(What:="*", _
                                 After:=ws.Cells(1, 1), _
                                 LookIn:=xlFormulas, _
                                 LookAt:=xlPart, _
                                 SearchOrder:=xlByColumns, _
                                 SearchDirection:=xlPrevious)
    On Error GoTo 0
    
    If lastCell Is Nothing Then
        GetLastUsedCol = 0
    Else
        GetLastUsedCol = lastCell.Column
    End If

End Function

Private Sub CopyColumnWidths(ByVal wsCalc As Worksheet, ByVal wsValidation As Worksheet)

    Dim lastCol As Long
    Dim c As Long
    
    lastCol = GetLastUsedCol(wsCalc)
    If lastCol = 0 Then Exit Sub
    
    For c = 1 To lastCol
        wsValidation.Columns(c).ColumnWidth = wsCalc.Columns(c).ColumnWidth
    Next c

End Sub

Private Sub CopyRowHeights(ByVal wsCalc As Worksheet, ByVal wsValidation As Worksheet)

    Dim lastRow As Long
    Dim r As Long
    
    lastRow = GetLastUsedRow(wsCalc)
    If lastRow = 0 Then Exit Sub
    
    For r = 1 To lastRow
        wsValidation.Rows(r).RowHeight = wsCalc.Rows(r).RowHeight
    Next r

End Sub
